# history / plan_to_watch (dokładnie tak samo, tylko nie ma pola data)
## wyświetlanie historii (opcjonalnie 10 i sortując)
SELECT movie.Title FROM history INNER JOIN movie ON history.Id_movie = movie.Id ORDER BY Date_view LIMIT 10;

# movie
## wyświetlanie tabeli filmów wraz z opcjonalnym szukaniem
SELECT * FROM movie WHERE Title <> '' AND Rating > 1 AND Year > 1915 OR Length > 20
## JOINY 

### tylko genre na movie
SELECT movie.Title, movie.Rating, movie.Id_writer, movie.Id_director, genre.Genre, movie.Year, movie.Length FROM movie INNER JOIN genre ON movie.Id_genre = genre.Id;

### genre i writer na movie
SELECT movie.Title, movie.Rating, writer.Writer, movie.Id_director, genre.Genre, movie.Year, movie.Length FROM movie INNER JOIN genre ON movie.Id_genre = genre.Id INNER JOIN writer ON movie.Id_writer = writer.Id;

### genre, writer i director na movie
SELECT movie.Title, movie.Rating, writer.Writer, director.Director, genre.Genre, movie.Year, movie.Length FROM movie INNER JOIN genre ON movie.Id_genre = genre.Id INNER JOIN writer ON movie.Id_writer = writer.Id INNER JOIN director ON movie.Id_director = director.Id;

### WHERE AND LIMIT
SELECT movie.Title, movie.Rating, writer.Writer, director.Director, genre.Genre, movie.Year, movie.Length FROM movie INNER JOIN genre ON movie.Id_genre = genre.Id INNER JOIN writer ON movie.Id_writer = writer.Id INNER JOIN director ON movie.Id_director = director.Id WHERE Director IN ('Géla Babluani') LIMIT 10;


# WEDŁUG OPISU Z MESSENGERA - baza users
## wyświetlanie danych o filmie z zawarciem gatunku, daty oglądania (dodania) i ratingiem użytkownika
SELECT movie.Title, review.Rate, movie.Year, movie.Length, genre.Genre, history.Date_view FROM movie INNER JOIN history ON movie.Id = history.Id_movie INNER JOIN review ON movie.Id = review.Id_movie INNER JOIN genre ON movie.Id_genre = genre.Id;
